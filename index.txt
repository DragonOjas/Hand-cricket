        let difficulty = "easy";
        let playerScore = 0, aiScore = 0, playerWickets = 0, aiWickets = 0;
        let currentInnings = 1, isPlayerBatting = true, gameFinished = false;
        let tossChoice = null, userTossNumber = null, aiTossNumber = null;
        let runOptions = [2, 4, 10, 15, 20];

        // ====================== SAVE & LOAD ======================
        function loadSavedData() {
            const savedName = localStorage.getItem("handCricket_playerName");
            if (savedName) playerName = savedName;

            const savedTheme = localStorage.getItem("handCricket_theme");
            if (savedTheme !== null) currentThemeIndex = parseInt(savedTheme);

            const savedStats = localStorage.getItem("handCricket_stats");
            if (savedStats) stats = JSON.parse(savedStats);

            document.getElementById("player-name-display").textContent = playerName;
            document.getElementById("lb-player-name").textContent = playerName;
        }

        function saveData() {
            localStorage.setItem("handCricket_playerName", playerName);
            localStorage.setItem("handCricket_theme", currentThemeIndex);
            localStorage.setItem("handCricket_stats", JSON.stringify(stats));
        }

        // ====================== BACKGROUND STARS ======================
        function createStars() {
            const container = document.getElementById("stars-container");
            for (let i = 0; i < 180; i++) {
                const star = document.createElement("div");
                star.className = "star";
                const size = Math.random() * 4 + 1.5;
                star.style.width = `${size}px`;
                star.style.height = `${size}px`;
                star.style.left = `${Math.random() * 100}%`;
                star.style.top = `${Math.random() * 100}%`;
                star.style.animationDelay = `-${Math.random() * 3}s`;
                star.style.opacity = Math.random();
                container.appendChild(star);
            }
        }

        // ====================== THEME ======================
        function applyTheme(index) {
            currentThemeIndex = index;
            const body = document.body;

            if (index === 0) {
                body.style.background = "linear-gradient(180deg, #0a0a2e 0%, #000011 100%)";
                document.getElementById("theme-icon").innerHTML = "🌊";
            } else if (index === 1) {
                body.style.background = "linear-gradient(180deg, #0a3d0a 0%, #001a00 100%)";
                document.getElementById("theme-icon").innerHTML = "🌿";
            } else if (index === 2) {
                body.style.background = "linear-gradient(180deg, #2a0066 0%, #110033 100%)";
                document.getElementById("theme-icon").innerHTML = "🌈";
            } else if (index === 3) {
                body.style.background = "linear-gradient(180deg, #8c4a00 0%, #3d2000 100%)";
                document.getElementById("theme-icon").innerHTML = "🏜️";
            }
            saveData();
            hideThemeModal();
        }

        // ====================== MODALS ======================
        function showNameModal() {
            document.getElementById("name-modal").classList.remove("hidden");
            const input = document.getElementById("name-input");
            input.value = playerName;
            input.focus();
        }

        function hideNameModal() {
            document.getElementById("name-modal").classList.add("hidden");
        }

        function savePlayerName() {
            const input = document.getElementById("name-input").value.trim();
            if (input) {
                playerName = input;
                document.getElementById("player-name-display").textContent = playerName;
                document.getElementById("lb-player-name").textContent = playerName;
                saveData();
            }
            hideNameModal();
        }

        function showThemeModal() {
            document.getElementById("theme-modal").classList.remove("hidden");
        }

        function hideThemeModal() {
            document.getElementById("theme-modal").classList.add("hidden");
        }

        function showLeaderboard() {
            document.getElementById("lb-played").textContent = stats.played;
            document.getElementById("lb-wins").textContent = stats.won;
            document.getElementById("lb-losses").textContent = stats.lost;
            document.getElementById("lb-high").textContent = stats.highScore || 0;
            document.getElementById("lb-low").textContent = stats.lowScore === Infinity ? "0" : stats.lowScore;
            document.getElementById("leaderboard-modal").classList.remove("hidden");
        }

        function hideLeaderboard() {
            document.getElementById("leaderboard-modal").classList.add("hidden");
        }

        // ====================== SCREEN & GAME LOGIC ======================
        function hideAllScreens() {
            document.querySelectorAll(".screen").forEach(s => s.classList.add("hidden"));
        }

        function backToMenu() { hideAllScreens(); document.getElementById("menu-screen").classList.remove("hidden"); }
        function backToMenuFromToss() { hideAllScreens(); document.getElementById("menu-screen").classList.remove("hidden"); resetGameState(); }
        function showLevels() { hideAllScreens(); document.getElementById("levels-screen").classList.remove("hidden"); }

        function startGame(level) {
            difficulty = level;
            hideAllScreens();
            resetGameState();
            document.getElementById("toss-screen").classList.remove("hidden");
            document.getElementById("player-toss-name").textContent = playerName;
            document.getElementById("toss-throw-btn").classList.add("hidden");
            document.getElementById("toss-result").classList.add("hidden");
            document.getElementById("odd-btn").classList.remove("border-cyan-400", "bg-cyan-900/60");
            document.getElementById("even-btn").classList.remove("border-cyan-400", "bg-cyan-900/60");
            tossChoice = null;
        }

        function selectTossSide(side) {
            tossChoice = side;
            const oddBtn = document.getElementById("odd-btn");
            const evenBtn = document.getElementById("even-btn");
            oddBtn.classList.toggle("border-cyan-400", side === "odd");
            oddBtn.classList.toggle("bg-cyan-900/60", side === "odd");
            evenBtn.classList.toggle("border-cyan-400", side === "even");
            evenBtn.classList.toggle("bg-cyan-900/60", side === "even");
            document.getElementById("toss-throw-btn").classList.remove("hidden");
        }

        function startTossThrow() {
            if (!tossChoice) return;
            userTossNumber = runOptions[Math.floor(Math.random() * runOptions.length)];
            aiTossNumber = runOptions[Math.floor(Math.random() * runOptions.length)];
            
            const playerHand = document.getElementById("player-hand-toss");
            const aiHand = document.getElementById("ai-hand-toss");
            
            setTimeout(() => {
                playerHand.classList.add("reveal"); aiHand.classList.add("reveal");
                playerHand.textContent = userTossNumber; aiHand.textContent = aiTossNumber;
                
                const sum = userTossNumber + aiTossNumber;
                const isEvenSum = sum % 2 === 0;
                const userWonToss = (tossChoice === "even" && isEvenSum) || (tossChoice === "odd" && !isEvenSum);
                
                const resultEl = document.getElementById("toss-result");
                resultEl.classList.remove("hidden");
                
                if (userWonToss) {
                    resultEl.innerHTML = `<span class="text-emerald-400">🎉 YOU WON THE TOSS!</span><br><span class="text-xl text-white/80 mt-6 block">Choose to BAT or BOWL</span><div class="flex justify-center gap-8 mt-8"><button onclick="chooseBatBowl(true)" class="px-12 py-5 bg-emerald-500 hover:bg-emerald-600 text-xl font-bold rounded-3xl">BAT FIRST</button><button onclick="chooseBatBowl(false)" class="px-12 py-5 bg-red-500 hover:bg-red-600 text-xl font-bold rounded-3xl">BOWL FIRST</button></div>`;
                } else {
                    const aiChoosesBat = Math.random() > 0.5;
                    resultEl.innerHTML = `<span class="text-red-400">CPU WON THE TOSS!</span><br><span class="text-xl text-white/80 mt-6 block">CPU elected to ${aiChoosesBat ? "BAT" : "BOWL"} first</span>`;
                    setTimeout(() => { isPlayerBatting = !aiChoosesBat; startInnings(); }, 2000);
                }
            }, 700);
        }

        function chooseBatBowl(bat) { isPlayerBatting = bat; startInnings(); }

        function startInnings() {
            hideAllScreens();
            document.getElementById("gameplay-screen").classList.remove("hidden");
            playerScore = aiScore = playerWickets = aiWickets = 0;
            currentInnings = 1; gameFinished = false;
            updateScoreboard();
            renderChoiceButtons();
            const statusEl = document.getElementById("game-status");
            statusEl.innerHTML = isPlayerBatting ? `<span class="text-cyan-400">YOU ARE BATTING</span>` : `<span class="text-red-400">YOU ARE BOWLING • AI BATTING</span>`;
            resetHands();
        }

        function updateScoreboard() {
            document.getElementById("player-score-name").textContent = playerName;
            document.getElementById("player-score").textContent = playerScore;
            document.getElementById("ai-score").textContent = aiScore;
            const wicketsEl = document.getElementById("wickets");
            wicketsEl.innerHTML = `<span class="text-cyan-400">YOU ${playerWickets}/1</span><span class="mx-4 text-white/30">•</span><span class="text-red-400">CPU ${aiWickets}/1</span>`;
            const statusEl = document.getElementById("innings-status");
            statusEl.textContent = currentInnings === 1 ? (isPlayerBatting ? "YOUR INNINGS" : "CPU INNINGS") : "2ND INNINGS";
        }

        function renderChoiceButtons() {
            const panel = document.getElementById("choice-panel");
            panel.innerHTML = "";
            runOptions.forEach(runs => {
                const btn = document.createElement("button");
                btn.className = `button-hover button-glow flex flex-col items-center justify-center bg-gradient-to-br from-cyan-400 to-emerald-400 hover:from-cyan-300 hover:to-emerald-300 text-black rounded-3xl py-8 text-3xl font-bold shadow-2xl hover:scale-105`;
                btn.innerHTML = `<div class="text-6xl mb-2">🏏</div><div>${runs}</div><div class="text-xs text-black/70 tracking-widest">RUNS</div>`;
                btn.onclick = () => handlePlayerChoice(runs);
                panel.appendChild(btn);
            });
        }

        function handlePlayerChoice(userChoice) {
            if (gameFinished) return;
            const playerHandEl = document.getElementById("player-hand");
            const playerDisplay = document.getElementById("player-choice-display");
            playerHandEl.textContent = "✊";
            playerDisplay.classList.remove("text-transparent");
            playerDisplay.textContent = userChoice;
            playerDisplay.style.color = "#22d3ee";
            
            let aiChoice;
            if (isPlayerBatting) {
                const matchChance = difficulty === "easy" ? 0.2 : (difficulty === "medium" ? 0.4 : 0.65);
                aiChoice = Math.random() < matchChance ? userChoice : runOptions[Math.floor(Math.random() * runOptions.length)];
            } else {
                aiChoice = runOptions[Math.floor(Math.random() * runOptions.length)];
            }
            
            setTimeout(() => {
                const aiHandEl = document.getElementById("ai-hand");
                const aiDisplay = document.getElementById("ai-choice-display");
                aiHandEl.classList.add("reveal");
                aiHandEl.textContent = aiChoice;
                aiDisplay.classList.remove("text-transparent");
                aiDisplay.textContent = aiChoice;
                aiDisplay.style.color = "#f87171";
                resolveBall(userChoice, aiChoice);
            }, 450);
        }

        function resolveBall(playerNum, aiNum) {
            const statusEl = document.getElementById("game-status");
            if (playerNum === aiNum) {
                statusEl.innerHTML = `<span class="out-text text-red-500 text-5xl">OUT! 🏏</span>`;
                if (isPlayerBatting) playerWickets++; else aiWickets++;
                updateScoreboard();
                if ((isPlayerBatting && playerWickets >= 1) || (!isPlayerBatting && aiWickets >= 1)) {
                    currentInnings++;
                    if (currentInnings > 2) { endMatch(); return; }
                    isPlayerBatting = !isPlayerBatting;
                    setTimeout(() => {
                        statusEl.innerHTML = isPlayerBatting ? `<span class="text-cyan-400">YOUR INNINGS STARTS!</span>` : `<span class="text-red-400">CPU INNINGS STARTS!</span>`;
                        resetHands(); updateScoreboard();
                    }, 1800);
                    return;
                }
            } else {
                const runsScored = isPlayerBatting ? playerNum : aiNum;
                if (isPlayerBatting) playerScore += runsScored; else aiScore += runsScored;
                statusEl.innerHTML = `<span class="score-pop text-emerald-400">+${runsScored} RUNS 🔥</span>`;
                updateScoreboard();
                if (runsScored >= 10) createSimpleConfetti();
            }
            setTimeout(() => { resetHands(); if (!gameFinished) statusEl.textContent = ""; }, 1700);
        }

        function resetHands() {
            document.getElementById("player-hand").textContent = "✊";
            document.getElementById("ai-hand").textContent = "✊";
            document.getElementById("player-choice-display").classList.add("text-transparent");
            document.getElementById("ai-choice-display").classList.add("text-transparent");
        }

        function endMatch() {
            gameFinished = true;
            stats.played++;
            const playerWon = playerScore > aiScore;
            const tie = playerScore === aiScore;
            if (playerWon) stats.won++; else if (!tie) stats.lost++;
            if (playerScore > stats.highScore) stats.highScore = playerScore;
            if (playerScore < stats.lowScore || stats.lowScore === Infinity) stats.lowScore = playerScore;
            saveData();
            hideAllScreens();
            const resultScreen = document.getElementById("result-screen");
            resultScreen.classList.remove("hidden");
            document.getElementById("final-player-score").textContent = playerScore;
            document.getElementById("final-ai-score").textContent = aiScore;
            const emojiEl = document.getElementById("result-emoji");
            const titleEl = document.getElementById("result-title");
            const subEl = document.getElementById("result-subtitle");
            if (tie) {
                emojiEl.textContent = "🤝"; titleEl.textContent = "IT'S A TIE!"; titleEl.style.color = "#eab308"; subEl.textContent = "Amazing match!";
            } else if (playerWon) {
                emojiEl.textContent = "🏆"; titleEl.innerHTML = `VICTORY<br><span class="text-cyan-400">${playerName}</span>`; subEl.textContent = "You crushed the CPU!";
            } else {
                emojiEl.textContent = "🤖"; titleEl.innerHTML = `DEFEAT<br><span class="text-red-400">CPU WINS</span>`; subEl.textContent = "Better luck next time!";
            }
        }

        function playAgain() { hideAllScreens(); startGame(difficulty); }
        function quitGame() {
            if (confirm("Quit the current match?")) {
                hideAllScreens();
                document.getElementById("menu-screen").classList.remove("hidden");
                resetGameState();
            }
        }
        function resetGameState() {
            playerScore = aiScore = playerWickets = aiWickets = 0;
            gameFinished = false;
        }

        function createSimpleConfetti() {
            for (let i = 0; i < 80; i++) {
                setTimeout(() => {
                    const confetti = document.createElement("div");
                    confetti.textContent = ["🏏", "🔥", "✨", "🌟"][Math.floor(Math.random() * 4)];
                    confetti.style.position = "fixed";
                    confetti.style.left = Math.random() * 100 + "vw";
                    confetti.style.top = "-30px";
                    confetti.style.fontSize = "24px";
                    confetti.style.zIndex = "9999";
                    document.body.appendChild(confetti);
                    const duration = Math.random() * 2500 + 2000;
                    confetti.animate([{ transform: `translateY(0) rotate(0deg)`, opacity: 1 }, { transform: `translateY(${window.innerHeight + 100}px) rotate(${Math.random() * 800}deg)`, opacity: 0 }], { duration: duration, easing: "cubic-bezier(0.25, 0.1, 0.25, 1)" });
                    setTimeout(() => confetti.remove(), duration);
                }, i * 4);
            }
        }

        // ====================== INITIALIZATION ======================
        function initializeGame() {
            loadSavedData();
            createStars();
            applyTheme(currentThemeIndex);
            
            console.log("%c🚀 Hand Cricket - MindFlip Edition Loaded with Epic Stars & Animations!", "color:#00f5ff; font-size:15px; font-weight:bold");
            
            document.addEventListener('keydown', (e) => {
                if (e.key.toLowerCase() === "t") {
                    if (!document.getElementById("menu-screen").classList.contains("hidden")) showLevels();
                }
            });
        }

        window.onload = initializeGame;
    </script>
